// RV32 LUI和AUIPC指令测试文件
// 格式：32位十六进制机器码 // [地址] 汇编指令 ; 注释说明
// 可被$readmemh读取

// ============== 初始化阶段 ==============
00100093 // [0x00000000] addi x1, x0, 1   ; 初始化 x1 = 1 (测试计数器)
00000113 // [0x00000004] addi x2, x0, 0   ; 初始化 x2 = 0 (错误标志)
00000193 // [0x00000008] addi x3, x0, 0   ; 初始化 x3 = 0 (结果寄存器)

// ============== 测试1：基本LUI指令 ==============
123450b7 // [0x0000000C] lui x1, 0x12345  ; x1 = 0x12345000
00400113 // [0x00000010] addi x2, x0, 4   ; x2 = 4 (继续执行标志)
// 验证：x1应该为0x12345000

// ============== 测试2：LUI最大值测试 ==============
7ffff0b7 // [0x00000014] lui x1, 0x7ffff  ; x1 = 0x7ffff000 (最大正数)
00800113 // [0x00000018] addi x2, x0, 8   ; x2 = 8
// 验证：x1应该为0x7ffff000

// ============== 测试3：LUI最小值测试 ==============
800000b7 // [0x0000001C] lui x1, 0x80000  ; x1 = 0x80000000 (最小负数)
00c00113 // [0x00000020] addi x2, x0, 12  ; x2 = 12
// 验证：x1应该为0x80000000

// ============== 测试4：LUI零值测试 ==============
000000b7 // [0x00000024] lui x1, 0x0      ; x1 = 0x00000000
01000113 // [0x00000028] addi x2, x0, 16  ; x2 = 16
// 验证：x1应该为0x00000000

// ============== 测试5：LUI负数测试 ==============
fffff2b7 // [0x0000002C] lui x5, 0xfffff  ; x5 = 0xfffff000 (负数扩展)
01400113 // [0x00000030] addi x2, x0, 20  ; x2 = 20
// 验证：x5应该为0xfffff000

// ============== 测试6：LUI与ADDI配合构建完整立即数 ==============
12345337 // [0x00000034] lui x6, 0x12345  ; x6 = 0x12345000
67830313 // [0x00000038] addi x6, x6, 0x678 ; x6 = 0x12345678
01800113 // [0x0000003C] addi x2, x0, 24  ; x2 = 24
// 验证：x6应该为0x12345678

// ============== 测试7：基本AUIPC指令 ==============
// 注意：AUIPC指令在地址0x40执行
// PC = 0x40, 立即数 = 0x1000
01000197 // [0x00000040] auipc x3, 0x1000  ; x3 = 0x40 + 0x1000000 = 0x1000040
01c00113 // [0x00000044] addi x2, x0, 28  ; x2 = 28
// 验证：x3应该为0x1000040

// ============== 测试8：AUIPC负数测试 ==============
fffff197 // [0x00000048] auipc x3, 0xfffff ; x3 = 0x48 + 0xfffff000 = 0xfffff048
02000113 // [0x0000004C] addi x2, x0, 32  ; x2 = 32
// 验证：x3应该为0xfffff048
